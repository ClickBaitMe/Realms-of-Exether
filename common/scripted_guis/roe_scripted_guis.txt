cast_ritual_button_effect = {
	saved_scopes = { ritual_id }
	effect = {
		add_modifier = {
			name = country_ritual_cooldown
			days = 1
		}
		pick_ritual = yes
		pick_ritual_cost = yes
	}
	is_valid = {
		AND = {
			NOT = { 
				has_modifier = country_ritual_cooldown
			}
			ritual_is_valid = yes
			scope:ritual_id = {
				custom_tooltip = {
					text = has_reserves_tt
					root.magic_reserves_total >= ritual_cost
				}
			}
		}
       
    }
}
