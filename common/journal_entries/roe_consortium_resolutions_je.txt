# triggered by CONSORTIUM Resolution Screen
je_consortium_declaration_of_human_rights = {
	icon = "gfx/interface/icons/cwp_icons/united_nations.dds"
	group = je_consortium_group_foreign_affairs
	should_be_pinned_by_default = yes
	possible = {
		OR = {
			has_variable = consortium_member
			has_variable = consortium_major_member
		}
		NOR = {
			has_global_variable = declaration_of_human_rightscomplete
			has_global_variable = declaration_of_human_rightsfail
			has_global_variable = vetoed_consortium
		}
	}
	current_value = {
		value = pending_resolution_progress_value_consortium
	}
	goal_add_value = {
		add = 100		# was 100, temp testing value
	}
	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					has_variable = consortium_major_member
				}
				change_global_variable = {
					name = pending_resolution_progress_consortium
					add = 0.5
				}
			}
		}
	}
	immediate = {
		every_country = {
			limit = {
				has_variable = consortium_member
			}
			save_scope_as = s_consortium_memberstates
		}
	}
	##### TODO, #32 Journal completion does NOT trigger event
	complete = {
		and = {
			scope:journal_entry = {
				is_goal_complete = yes
			}
			custom_tooltip = {
				text = "WILL_COMPLETE_CONSORTIUM"
				supporting_value_total_consortium > opposing_value_total_consortium
			}
		}
	}
	fail = {
		or = {
			AND = {
				scope:journal_entry = {
					is_goal_complete = yes
				}
				custom_tooltip = {
					text = "WILL_FAIL_CONSORTIUM"
					opposing_value_total_consortium > supporting_value_total_consortium
				}
			}
			custom_tooltip = {
				text = "IS_VETOED"
				has_global_variable = vetoed_consortium
			}
		}
	}
	on_complete = {
		hidden_effect = {
			every_country = {
				trigger_event = {
					id = consortium_resolution_events.1
				}
			}
			every_country = {
				limit = {
					has_variable = consortium_member
				}
				add_modifier = {
					name = declaration_of_human_rights
				}
			}
		}
		effect_reset_consortium_lists = yes
		set_global_variable = declaration_of_human_rightscomplete
	}
	on_fail = {
		hidden_effect = {
			every_country = {
				trigger_event = {
					id = consortium_resolution_events.2
				}
			}
		}
		effect_reset_consortium_lists = yes
		set_global_variable = {
			name = declaration_of_human_rightsfail
			days = 365
		}
	}
	progressbar = yes
	can_deactivate = yes
	weight = 100
}
